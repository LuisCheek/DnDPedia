CREATE TABLE "SPELLS" (
	"name"	TEXT NOT NULL,
	"source"	TEXT NOT NULL,
	"level"	INTEGER NOT NULL DEFAULT 0 CHECK("level" >= 0 AND "level" <= 9),
	"school"	TEXT NOT NULL,
	"classes"	TEXT NOT NULL,
	"verbal"	INTEGER NOT NULL DEFAULT 0 CHECK("verbal" IN (0, 1)),
	"somatic"	INTEGER NOT NULL DEFAULT 0 CHECK("somatic" IN (0, 1)),
	"material"	INTEGER NOT NULL DEFAULT 0 CHECK("material" IN (0, 1)),
	"materials"	TEXT,
	"time"	TEXT NOT NULL,
	"area"	TEXT,
	"range"	INTEGER NOT NULL DEFAULT 0,
	"duration"	TEXT NOT NULL,
	"concentration"	INTEGER NOT NULL DEFAULT 0 CHECK("concentration" IN (0, 1)),
	"ritual"	INTEGER NOT NULL DEFAULT 0 CHECK("ritual" IN (0, 1)),
	"spell"	TEXT NOT NULL,
	"attack"	INTEGER NOT NULL DEFAULT 0 CHECK("attack" IN (0, 1)),
	"damage"	TEXT,
	"save"	INTEGER NOT NULL DEFAULT 0 CHECK("save" IN (0, 1)),
	"saveType"	TEXT,
	"higherLevels"	INTEGER NOT NULL DEFAULT 0 CHECK("higherLevels" IN (0, 1)),
	"conditions"	TEXT,
	"tags"	TEXT,
	FOREIGN KEY("source") REFERENCES "SOURCES"("Source"),
	PRIMARY KEY("name")
);
